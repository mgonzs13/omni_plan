97 void declare_ros_parameters(rclcpp::Node::SharedPtr node) const {
98 for (const auto &param : this->params_) {
99 if (!param.declared) {
100 std::string full_name = this->namespace_ + "." + param.name;
101 if (!node->has_parameter(full_name)) {
102 node->declare_parameter(full_name, param.default_value);
103 }
104 param.declared = true;

121 void load_ros_parameters(rclcpp::Node::SharedPtr node) const {
122 this->declare_ros_parameters(node);
123
124 for (const auto &param : this->params_) {
125 this->load_single_ros_parameter(node, param);
126 }
127
128 for (const auto &cb : this->callbacks_) {

130 }

133private:
134 void load_single_ros_parameter(rclcpp::Node::SharedPtr node,

136 std::string full_name = this->namespace_ + "." + param.name;
137 rclcpp::ParameterValue val;
138 if (node->get_parameter(full_name, val)) {
139 param.setter(val);
140 } else {
141 param.setter(param.default_value);
142 }
143 }
