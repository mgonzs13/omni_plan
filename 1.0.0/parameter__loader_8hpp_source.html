97 void declare_ros_parameters(rclcpp::Node::SharedPtr node) const {

119 void load_ros_parameters(rclcpp::Node::SharedPtr node) const {
120 this->declare_ros_parameters(node);
121
122 for (const auto &param : this->params_) {
123 this->load_single_ros_parameter(node, param);
124 }
125
126 for (const auto &cb : this->callbacks_) {
127 cb();
128 }
129 }

131private:
132 void load_single_ros_parameter(rclcpp::Node::SharedPtr node,
133 const ParameterInfo &param) const {
134 std::string full_name = this->namespace_ + "." + param.name;
135 rclcpp::ParameterValue val;
136 if (node->get_parameter(full_name, val)) {
137 param.setter(val);
138 } else {
139 param.setter(param.default_value);
140 }
141 }
